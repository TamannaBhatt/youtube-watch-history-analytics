def load_watch_history(file_path):

    with open(
        file_path,
        "r",
        encoding="utf-8"
    ) as f:

        soup = BeautifulSoup(
            f,
            "html.parser"
        )

    blocks = soup.find_all("div")

    pattern = r"\d{1,2}\s\w+\s\d{4},\s\d{2}:\d{2}:\d{2}\sIST"

    records = []

    for block in blocks:

        try:

            text = block.get_text(
                " ",
                strip=True
            )

            if "Watched " not in text:
                continue

            links = block.find_all("a")

            if len(links) == 0:
                continue

            title = links[0].text.strip()

            match = re.search(
                pattern,
                text
            )

            if not match:
                continue

            watch_time = match.group()

            is_ad = "From Google Ads" in text

            records.append({

                "title": title,

                "watch_time": watch_time,

                "is_ad": is_ad,

                "raw_text": text

            })

        except Exception:
            continue

    df = pd.DataFrame(records)

    print("Raw Records :", len(df))

    return df

# ==========================================================
# Data Preprocessing
# ==========================================================

def preprocess_data(df):

    df.drop_duplicates(
        inplace=True
    )

    df["watch_time"] = pd.to_datetime(

        df["watch_time"],

        format="%d %B %Y, %H:%M:%S IST",

        errors="coerce"

    )

    df.dropna(
        inplace=True
    )

    return df


# ==========================================================
# Feature Engineering
# ==========================================================

def get_time_category(hour):

    if 5 <= hour < 12:
        return "Morning"

    elif 12 <= hour < 17:
        return "Afternoon"

    elif 17 <= hour < 21:
        return "Evening"

    return "Night"


def create_features(df):

    df["hour"] = df["watch_time"].dt.hour

    df["day"] = df["watch_time"].dt.day_name()

    df["date"] = df["watch_time"].dt.date

    df["time_category"] = df["hour"].apply(
        get_time_category
    )

    return df

# Hourly Activity Analysis
# ==========================================================

def hourly_activity_analysis(df):

    hour_counts = (
        df["hour"]
        .value_counts()
        .sort_index()
    )

    return hour_counts


# ==========================================================
# Statistical Anomaly Detection
# ==========================================================

def detect_anomalies(hour_counts):

    mean_activity = hour_counts.mean()

    std_activity = hour_counts.std()

    threshold = mean_activity + std_activity

    anomalies = hour_counts[
        hour_counts > threshold
    ]

    print("\nHourly Activity\n")

    print(hour_counts)

    print("\nStatistical Anomalies\n")

    print(anomalies)

    print("\nSummary\n")

    print(f"Peak Activity Hour       : {hour_counts.idxmax()}")

    print(f"Maximum Videos Watched   : {hour_counts.max()}")

    print(f"Anomalous Hours          : {list(anomalies.index)}")

    return threshold, anomalies

def categorize_video(title):

    title = title.lower()

    for category, keywords in CATEGORY_KEYWORDS.items():

        for keyword in keywords:

            if keyword.lower() in title:

                return category

    return "Others"


def add_content_categories(df):

    df["category"] = df["title"].apply(
        categorize_video
    )

    return df

# ==========================================================
# Day vs Hour Heatmap
# ==========================================================

def plot_heatmap(df):

    day_order = [

        "Monday",
        "Tuesday",
        "Wednesday",
        "Thursday",
        "Friday",
        "Saturday",
        "Sunday"

    ]

    heatmap_data = df.pivot_table(

        index="day",

        columns="hour",

        values="title",

        aggfunc="count",

        fill_value=0

    )

    heatmap_data = heatmap_data.reindex(day_order)

    plt.figure(figsize=(14,6))

    sns.heatmap(

        heatmap_data,

        annot=True,

        fmt="d",

        cmap="YlOrRd",

        linewidths=0.5

    )

    plt.title("Day vs Hour Activity Heatmap")

    plt.tight_layout()

    plt.savefig(

        os.path.join(

            OUTPUT_DIR,

            "activity_heatmap.png"

        ),

        dpi=300

    )

    plt.show()

    text = " ".join(

        df["title"].astype(str)

    )

        width=1200,

        background_color="white",

        collocations=False

        wordcloud,

# Export Functions
# ==========================================================

def save_cleaned_dataset(df):

    cleaned_csv = os.path.join(
        OUTPUT_DIR,
        "youtube_watch_history_cleaned.csv"
    )

    df.to_csv(
        cleaned_csv,
        index=False
    )

    print(f"\nCleaned dataset saved to: {cleaned_csv}")


def save_analysis_summary(df,
                          hour_counts,
                          anomalies,
                          weekly_activity):

    summary_file = os.path.join(
        OUTPUT_DIR,
        "analysis_summary.txt"
    )

    with open(
        summary_file,
        "w",
        encoding="utf-8"
    ) as f:

        f.write("YouTube Watch History Analysis Summary\n")
        f.write("=" * 50 + "\n\n")

        f.write(f"Total Watch Records : {len(df)}\n")
        f.write(f"Normal Videos       : {(~df['is_ad']).sum()}\n")
        f.write(f"Advertisements      : {df['is_ad'].sum()}\n\n")

        f.write(f"Peak Activity Hour  : {hour_counts.idxmax()}\n")
        f.write(f"Peak Activity Day   : {weekly_activity.idxmax()}\n")
        f.write(f"Most Common Category: {df['category'].value_counts().idxmax()}\n\n")

        f.write("Anomalous Hours\n")
        f.write("--------------------------\n")

        if len(anomalies) > 0:

            for hour, count in anomalies.items():

                f.write(
                    f"Hour {hour}: {count} videos\n"
                )

        else:

            f.write(
                "No anomalies detected.\n"
            )

    print(f"Summary saved to: {summary_file}")


def save_reports(top_ads,
                 anomalies):

    top_ads.to_csv(
        os.path.join(
            OUTPUT_DIR,
            "top_advertisements.csv"
        )
    )

    anomalies.to_csv(
        os.path.join(
            OUTPUT_DIR,
            "anomaly_report.csv"
        )
    )

# ==========================================================
# Main Function
# ==========================================================

def main():

    # ------------------------
    # Load Dataset
    # ------------------------

    df = load_watch_history(FILE_PATH)

    # ------------------------
    # Preprocess
    # ------------------------

    df = preprocess_data(df)

    df = create_features(df)

    df = add_content_categories(df)

    # ------------------------
    # Analysis
    # ------------------------

    dataset_summary(df)

    hour_counts = hourly_activity_analysis(df)

    threshold, anomalies = detect_anomalies(
        hour_counts
    )

    weekly_activity = weekly_activity_analysis(
        df
    )

    top_ads = top_ads_analysis(df)

    active_dates = top_active_dates(df)

    # ------------------------
    # Visualizations
    # ------------------------

    plot_hourly_activity(
        hour_counts,
        threshold
    )

    plot_weekly_activity(
        weekly_activity
    )

    plot_heatmap(df)

    plot_watch_distribution(df)

    plot_ad_distribution(df)

    plot_advertisement_by_time(df)

    plot_top_ads(top_ads)

    plot_top_dates(
        active_dates
    )

    plot_content_categories(df)

    generate_wordcloud(df)

    # ------------------------
    # Export Results
    # ------------------------

    save_cleaned_dataset(df)

    save_analysis_summary(

        df,

        hour_counts,

        anomalies,

        weekly_activity

    )

    save_reports(
    top_ads,
    anomalies
    )

    print("\n" + "=" * 60)

    print("Analysis Completed Successfully")

    print("=" * 60)


if __name__ == "__main__":
    main()
